182  private void Move(string move)
183  {
184  if (!this.gameLogic.GameOver())

189  case "RIGHT": this.gameLogic.MoveLogic.MoveDrill(1, 0); break;

193  }

227  case Key.F1: this.gameLogic.MoveLogic.UpgradeDrill(); break;
228  case Key.F2: this.gameLogic.MoveLogic.UpgradeStorage(); break;
229  case Key.F3: this.gameLogic.MoveLogic.UpgradeFuelTank(); break;
