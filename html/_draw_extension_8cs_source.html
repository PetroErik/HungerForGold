190  private Brush GetBrush(string fname, bool isTiled)
191  {
192  if (!this.brushes.ContainsKey(fname))
193  {
194  BitmapImage img = new BitmapImage();
195  img.BeginInit();
196  img.StreamSource = Assembly.GetExecutingAssembly().GetManifestResourceStream(fname);
197  img.EndInit();
198  ImageBrush ib = new ImageBrush(img);
199  if (isTiled)
200  {
201  ib.TileMode = TileMode.Tile;
202  ib.Viewport = new Rect(0, 0, this.width, this.height);
203  ib.ViewportUnits = BrushMappingMode.Absolute;
204  }
205 
206  this.brushes.Add(fname, ib);
207  }
208 
209  return this.brushes[fname];
210  }
