99  double minY = this.gameModel.Drill.Location[1] - 10;
100  double maxY = minY + this.gameModel.TileSize;
101 
102  Random r = new Random();
103 
104  if (minX <= min.Location[0] && maxX >= min.Location[0] && minY <= min.Location[1] && maxY >= min.Location[1]
105  && this.gameModel.Drill.StorageFullness < this.gameModel.Drill.StorageCapacity)
106  {
107  switch (min.Type)
108  {
109  case MineralsType.Gold:
110  this.gameModel.ActualPoints += CONFIG.GoldPrice; this.gameModel.Drill.StorageFullness++;
111  break;
112  case MineralsType.Silver:
113  this.gameModel.ActualPoints += CONFIG.SilverPrice; this.gameModel.Drill.StorageFullness++;
114  break;
115  case MineralsType.Bronze:
116  this.gameModel.ActualPoints += CONFIG.BronzePrice; this.gameModel.Drill.StorageFullness++;
117  break;
118  }
119 
120  min.Location[0] = r.Next(0, CONFIG.MapWidth * 2) * this.gameModel.TileSize;
121  min.Location[1] = r.Next((CONFIG.MapHeight / 3) + 2, CONFIG.MapHeight) * this.gameModel.TileSize;
122  }
